\chapter{Dữ liệu tự chọn}
\begin{itemize}
	\item Tên "đề tài", nguồn gốc của dữ liệu, giới thiệu các biến.
	\item Mô hình chọn được; phân tích kết quả
	\item Đưa ra những phương pháp/phân tích khác có thể giúp cho kết quả tốt hơn.
	\item Kết luận.
\end{itemize}
